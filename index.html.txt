// ── NAV ───────────────────────────────────────────────
function showPage(id) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active'));
  document.getElementById('page-' + id).classList.add('active');
  document.querySelectorAll('.nav-tab').forEach(t => {
    if (t.textContent.toLowerCase().includes(id.substring(0,4))) t.classList.add('active');
  });
  if (id === 'tracker') renderTracker();
  if (id === 'reminder') renderReminder();
}

// ── DATE ──────────────────────────────────────────────
function getDayInfo() {
  const start = new Date('2026-06-12'); // battle plan start
  const now = new Date();
  const diff = Math.floor((now - start) / 86400000);
  const day = Math.max(1, Math.min(30, diff + 1));
  return { day, now };
}

function formatDate(d) {
  return d.toLocaleDateString('en-IN', { weekday: 'long', day: 'numeric', month: 'long' });
}

// ── WEEK TOGGLE ───────────────────────────────────────
function toggleWeek(header) {
  header.parentElement.classList.toggle('open');
}

// ── TRACKER ───────────────────────────────────────────
function addApplication() {
  const company = document.getElementById('f-company').value.trim();
  const role = document.getElementById('f-role').value.trim();
  if (!company || !role) { alert('Company and role are required.'); return; }
  apps.unshift({
    id: Date.now(),
    company,
    role,
    platform: document.getElementById('f-platform').value,
    status: document.getElementById('f-status').value,
    notes: document.getElementById('f-notes').value.trim(),
    date: new Date().toLocaleDateString('en-IN', { day: 'numeric', month: 'short' })
  });
  saveApps();
  document.getElementById('f-company').value = '';
  document.getElementById('f-role').value = '';
  document.getElementById('f-notes').value = '';
  renderTracker();
}

function deleteApp(id) {
  apps = apps.filter(a => a.id !== id);
  saveApps();
  renderTracker();
}

function updateStatus(id, val) {
  const a = apps.find(a => a.id === id);
  if (a) { a.status = val; saveApps(); renderTracker(); }
}

const statusMap = {
  applied: ['Applied', 's-applied'],
  interview: ['Interview', 's-interview'],
  offer: ['Offer 🎉', 's-offer'],
  rejected: ['Rejected', 's-rejected'],
  followup: ['Follow up', 's-followup'],
};

function renderTracker() {
  const counts = { total: apps.length, interview: 0, offer: 0, rejected: 0 };
  apps.forEach(a => {
    if (a.status === 'interview') counts.interview++;
    if (a.status === 'offer') counts.offer++;
    if (a.status === 'rejected') counts.rejected++;
  });

  document.getElementById('tracker-stats').innerHTML = `
    <div class="t-stat"><div class="t-stat-num" style="color:var(--accent)">${counts.total}</div><div class="t-stat-label">Applied</div></div>
    <div class="t-stat"><div class="t-stat-num" style="color:var(--amber)">${counts.interview}</div><div class="t-stat-label">Interviews</div></div>
    <div class="t-stat"><div class="t-stat-num" style="color:var(--green)">${counts.offer}</div><div class="t-stat-label">Offers</div></div>
    <div class="t-stat"><div class="t-stat-num" style="color:var(--red)">${counts.rejected}</div><div class="t-stat-label">Rejected</div></div>
  `;

  const pct = Math.min(100, Math.round((counts.total / 150) * 100));
  document.getElementById('progress-fill').style.width = pct + '%';
  document.getElementById('progress-label-text').textContent = `${counts.total} / 150`;

  const tbody = document.getElementById('apps-tbody');
  if (apps.length === 0) {
    tbody.innerHTML = '<tr><td colspan="8" class="empty-state">No applications yet. Log your first one above ↑</td></tr>';
    return;
  }
  tbody.innerHTML = apps.map((a, i) => {
    const [label, cls] = statusMap[a.status] || ['Applied', 's-applied'];
    return `<tr>
      <td style="color:var(--muted)">${apps.length - i}</td>
      <td style="font-weight:500">${a.company}</td>
      <td style="color:var(--muted)">${a.role}</td>
      <td style="color:var(--muted)">${a.platform}</td>
      <td style="color:var(--muted);white-space:nowrap">${a.date}</td>
      <td>
        <select onchange="updateStatus(${a.id}, this.value)" style="background:transparent;border:none;color:inherit;font-size:11px;cursor:pointer;outline:none;">
          ${Object.entries(statusMap).map(([v,[l,c]]) =>
            `<option value="${v}" ${a.status===v?'selected':''} style="background:#111827">${l}</option>`
          ).join('')}
        </select>
        <span class="status-badge ${cls}">${label}</span>
      </td>
      <td style="color:var(--muted);font-size:11px;max-width:140px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap">${a.notes || '—'}</td>
      <td><button class="btn-del" onclick="deleteApp(${a.id})">×</button></td>
    </tr>`;
  }).join('');
}

function renderReminder() {
  const { day, now } = getDayInfo();
  const week = Math.min(4, Math.ceil(day / 7));
  const todayKey = now.toDateString();
  const saved = dailyChecks[todayKey] || [];

  document.getElementById('mission-date').textContent = `Day ${day} of 30 · ${formatDate(now)}`;
  document.getElementById('mission-title').textContent =
    week === 1 ? 'Week 1: Foundations & first wave' :
    week === 2 ? 'Week 2: Full volume engine on' :
    week === 3 ? 'Week 3: Interviews + keep applying' :
    'Week 4: Convert & negotiate';

  const tasks = weekMissions[week];
  document.getElementById('mission-tasks').innerHTML = tasks.map((t, i) => {
    const done = saved.includes(i);
    return `<div class="mission-task ${done ? 'done' : ''}" onclick="toggleCheck(${i}, '${todayKey}')">
      <div class="mission-check">${done ? '✓' : ''}</div>
      <div class="mission-task-text">${t}</div>
    </div>`;
  }).join('');
}

function toggleCheck(idx, key) {
  if (!dailyChecks[key]) dailyChecks[key] = [];
  const arr = dailyChecks[key];
  const pos = arr.indexOf(idx);
  if (pos === -1) arr.push(idx); else arr.splice(pos, 1);
  saveChecks();
  renderReminder();
}

const reminderMessages = {
  morning: ['🌅 Morning check', 'Open LinkedIn and Naukri — check job alerts now.'],
  evening: ['📋 Apply time', 'Your evening application session starts now. Aim for 5 jobs.'],
  outreach: ['📩 Outreach time', 'Message 5 recruiters on LinkedIn. Keep it sharp and brief.'],

};

function toggleReminder(btn, key, hour) {
  btn.classList.toggle('on');
  if (btn.classList.contains('on')) {
    scheduleReminder(key, hour);
  }
}

function scheduleReminder(key, hour) {
  const now = new Date();
  const target = new Date();
  target.setHours(hour, 0, 0, 0);
  if (target <= now) target.setDate(target.getDate() + 1);
  const delay = target - now;
  setTimeout(() => showNotif(...reminderMessages[key]), delay);
}

function showNotif(title, body) {
  document.getElementById('notif-title').textContent = title;
  document.getElementById('notif-body').textContent = body;
  const banner = document.getElementById('notif-banner');
  banner.classList.add('show');
  setTimeout(() => banner.classList.remove('show'), 5000);

  if ('Notification' in window && Notification.permission === 'granted') {
    new Notification(title, { body, icon: '📋' });
  }
}
